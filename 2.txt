#include <stdio.h>
int main()
{
    int R,C;
    scanf("%d %d",&R,&C);
    int a[R][C];
    int maxProduct = -999999;
    for(int i=0;i<R;i++)
        for(int j=0;j<C;j++)
            scanf("%d",&a[i][j]);
            for(int i=0;i<C;i++)
    {
        for(int j=0;j<C;j++)
        {
            int product = a[0][i] * a[1][j];
            if(product > maxProduct)
                maxProduct = product;
        }
    }
    printf("%d",maxProduct);
}
